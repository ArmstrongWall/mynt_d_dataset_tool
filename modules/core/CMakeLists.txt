set(CURRENT_DIR ${CMAKE_CURRENT_SOURCE_DIR})

# mynteye_core

set(MYNTEYE_CORE ${PROJECT_NAME}_core)

set(MYNTEYE_CORE_INCLUDES
  ${CURRENT_DIR}/include
  ${CURRENT_DIR}/include/api/
)

include_directories(
  ${MYNTEYE_CORE_INCLUDES}
  ${OpenCV_INCLUDE_DIRS}
  ${eSPDI_INCLUDE_DIRS}
  ${JPEG_INCLUDE_DIR}
)

file(GLOB_RECURSE MYNTEYE_CORE_SRCS "src/*.cc")

make_shared_library(${MYNTEYE_CORE}
  SRCS
    ${MYNTEYE_CORE_SRCS}
  LINKLIBS
    ${OpenCV_LIBS} ${eSPDI_LIBS} ${JPEG_LIBRARIES}
)

set(MYNTEYE_CORE ${MYNTEYE_CORE} PARENT_SCOPE)
set(MYNTEYE_CORE_INCLUDES ${MYNTEYE_CORE_INCLUDES} PARENT_SCOPE)
